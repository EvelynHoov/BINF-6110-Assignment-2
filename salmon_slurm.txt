#!/usr/bin/env bash
#SBATCH --job-name=salmon_quant
#SBATCH --array=0-8
#SBATCH --cpus-per-task=8
#SBATCH --mem=32G
#SBATCH --time=04:00:00
#SBATCH --output=logs/%x_%A_%a.out
#SBATCH --error=logs/%x_%A_%a.err

set -euo pipefail

INDEX="salmon_index"
SAMPLES_TSV="SC_samples.tsv"
OUTDIR="salmon_quant"

mkdir -p logs

line="$(sed -n "$((SLURM_ARRAY_TASK_ID+1))p" "${SAMPLES_TSV}")"
sample="$(echo "${line}" | awk '{print $1}')"
r1="$(echo "${line}" | awk '{print $2}')"
r2="$(echo "${line}" | awk '{print $3}')"

if [[ -z "${sample}" || -z "${r1}" ]]; then
  echo "Bad/missing line in ${SAMPLES_TSV} for task ${SLURM_ARRAY_TASK_ID}" >&2
  exit 1
fi

# Single-end: r2 is "." or empty
salmon quant \
  -i "${INDEX}" \
  -l A \
  -r "${r1}" \
  -p "${SLURM_CPUS_PER_TASK}" \
  --validateMappings \
  -o "${OUTDIR}/${sample}"
